NUM_REELS = 3

class SpinRequest(BaseModel):
    bet: int = 1

def spin(symbols, weights, num_reels=NUM_REELS):
    if len(symbols) != len(weights):
        raise ValueError("Symbols and weights lists must have the same length.")

    population = []
    for i, symbol in enumerate(symbols):
        population.extend([symbol] * weights[i])

    reels = tuple(random.choice(population) for _ in range(num_reels))
    return reels

def calculate_payout(reels, payouts, bet_amount):
    num_reels = len(reels)

    # Check for exact matches
    if reels in payouts:
        return payouts[reels] * bet_amount

    # Check for wildcard-like matches (represented by _)
    for pattern, multiplier in payouts.items():
        if len(pattern) == num_reels:
            match = True
            for i in range(num_reels):
                if pattern[i] != "_" and pattern[i] != reels[i]:
                    match = False
                    break
            if match:
                return multiplier * bet_amount

    return 0

@app.route('/api/slots/spin', methods=['POST'])
